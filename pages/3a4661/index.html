第二步用实现的是，对resolve的使用（reject暂不管），还有，就是then的接收。

const isFunction = (v) => typeof v === "function";
const PENDING = "PENDING";
const FULFILLED = "FULFILLED";
const REJECTED = "REJECTED";
class MyPromise {
  constructor(handler) {
    if(!isFunction(handler)) throw new Error("promise参数不是一个函数")

显而易见，这样是不行的，因为then调用的时机问题，this.value的值不是最新的。那么，接下来的思路就是将then的调用存储起来，在resolve的时候再一起去使用。

于是，改造一下：

class MyPromise {
  constructor(handler) {
    if(!isFunction(handler)) throw new Error("promise参数不是一个函数")

    this.status = PENDING; // 用过promise就觉得，他是有状态的，resolve前十pending状态，resolve后是fulfilled，错误时是rejected
    this.value = undefined; // 用来存储，then的参数值
    this.fulfilledQueues = []; // 执行成功队列。
    try {
      handler(this.resolve()
    } catch (err) {
      // 暂留.
    }
  }
  resolve(value) {
	if(this.status !== PENDING) return; // 已经是ful或rej状态，自然不能执行了。
    //  依次执行成功队列，并清空队列
    const run = () =>{
      this.status = FULFILLED; // 将promise状态改为ful；
      this.value = value;
      // 将then函数放在赋值后调用，就能保证执行顺序了
      let cb;
      while(cb = this.fulfilledQueues.shift()){
        cb(value)
      }
    }
    run();
  }
  then(onFulfilled) { // 因为是要将状态改为Fulfilled，所以加上on前缀的起名好
    const { status } = this;  
    let fulfilled = (val) => {
      onFulfilled(val)
    }
    switch (status) {
      case PENDING:
        this.fulfilledQueues.push(fulfilled);
    }
  }
}

# then

then(onFulfilled, onRejected) { // 因为是要将状态改为Fulfilled，所以加上on前缀的起名好
  const { value, status } = this;  
  return new MyPromise((onFulfilledNext, onRejectedNext) => {
    let fulfilled = (val) => {
      try {
        if(typeof onFulfilled !== "function") {
          onFulfilledNext(val)
        } else {
          let res = onFulfilled(val); // 拿到返回值
          if(res instanceof MyPromise){
            res.then(onFulfilledNext) // 这么做之后，只有当promise这个返回resolve的时候，原本的promise后面跟着的then才会继续执行下去
          }else {
            onFulfilledNext(res);
          }
        }
      } catch(err) {
        onRejectedNext(err);
      }    
    }
    let rejected = (error) => {
      try {
        if (typeof onRejected !== "function") {
          onRejectedNext(error);
        } else {
          let res = onRejected(error);
          if (res instanceof MyPromise) {
            res.then(onFulfilledNext, onRejectedNext);
          } else {
            // 这里之所以这样，是因为源promise就是要实现执行reject后一个then要执行第一个函数的效果。
            onFulfilledNext(res);
          }
        }
      } catch (err) {
        onRejectedNext(err);
      }
    };
    switch (status) {
      case PENDING:
        this.fulfilledQueues.push(fulfilled);
        this.rejectedQueues.push(rejected);
        break;
      case FULFILLED:
        fulfilled(value);
        break;
      case REJECTED:
        rejected(value);
        break;
    }

  });
}

const run = () =>{
  this.status = FULFILLED; // 将promise状态改为ful；
  const runFufilled = (val) => {            
    let cb;
    while((cb = this.fulfilledQueues.shift())){
      cb(val)
    }
  }
  const runRejected = (val) => {            
    let cb;
    while((cb = this.rejectedQueues.shift())){
      cb(val)
    }
  }
  if(value instanceof MyPromise){
    value.then(
      (val) => {
        this.value = val;
        runFufilled(val);
      },
      (err) => {
        this.value = err;
        runRejected(val);
      }
    )
  }else {
    this.value = value;
    runFufilled(value);
  }
}

# all

这个功能就是等待所有的promise执行完毕(但凡有一个失败都不行)，再接着轮到他执行，参数要求是promise数组。全部正常执行完毕，就是返回每个promise返回值的数组，如果有失败的，那就是返回那个最先失败的promise的返回值

static all(list) {
  return new Promise((resolve, reject) => {
    let count = 0;
    let values = [];
    let len = list.length;
    for(let [i,pro] of list.entries()){ // 之所以要转为对象自身可枚举属性的键值对数组，是为了方便获取下标
      // 之所以不用count作为下标，是出于promise的执行成功不是按照参数顺序
      this.resolve(pro).then(
        (res) => {
          values[i] = res;
          count++;
          if(count === len) resolve(values);
        },
        (err) => {
          reject(err);
        }
      )
    }
  })
}

# race

与all相反，他只要当其中有某个promise成功或失败，那么就会执行下一步，获取的值是最先成功或失败的返回值

static race(list){
  return new MyPromise((resolve, reject) => {
    let isFinish = false;
    for(let pro of list){
      this.resolve(pro).then(
        (res) => {
          if(!isFinish){
            resolve(res)
            isFinish = true
          }
        },
        (err) => {
          if(!isFinish){
            reject(err)
            isFinish = true
          }
        }
      )
    }
  })
}

# finally

无论一个promise的运行时是什么结果，最终都会走入那个方法的回调函数。应用的时候比如一个弹窗有对有错，但最后都会关闭

finally(cb) {
  return this.then(
    (value) => MyPromise.resolve(cb()).then(() => value),
    (value) => MyPromise.resolve(cb()).then(() => value),
  )        
  // 这么写的用意是当调用这个之后，对应的其他功能应该还是能使用，并且要保证词穿透
}

# catch

可以说是then第二个参数的简写，让可看性好一点

catch(onRejected) {
  return new MyPromise((resolve, reject) => {
    const rejected = (error) => {
      try {
        if(typeof onRejected !== "function"){
          reject(error)
        } else {
          let res = onRejected(error);
          if(res instanceof MyPromise) {
            res.then(resolve, reject);
          }else {                                                
            reject(res);
          }
        }
      } catch(err) {
        reject(err)
      }
    }
    this.rejectedQueues.push(rejected)
  })
}
