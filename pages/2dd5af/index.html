# 冒泡排序

冒泡排序是最传统的一种排序方式，相信不需要过于废话，直接做了

    function bubleSort(arr){
      let len = arr.length;
      for(let i =len;i>1;i--)
        for(let j=0;j<=i-1;j++)
          if(arr[j]>arr[j+1])
          [arr[j],arr[j+1]] = [arr[j+1],arr[j]]

    }

# 选择排序

选择排序跟冒泡相反，他是从头开始，依次跟后面进行比较，将最小的放在前面，而冒泡就是将最大的放到最后

    function selectSort(arr){
      let len = arr.length;
      for(let i =0;i<len-1;i++)
        for(let j=i+1;j<len;j++)
          if(arr[i]>arr[j])
          [arr[i],arr[j]] = [arr[j],arr[i]]
    }

# 插入排序

插入排序的核心思想是扑克牌思维，什么意思呢，就是随机拿出一张卡片，然后看这张卡片跟前面的比谁比较小，小的就插入到前方（也就是互换），如此进行一个小循环后拿出另一张卡片以此类推

    function insertSort(arr){
      let len = arr.length;
      for(let i =1;i<len;i++)
        for(let j=i-1;j>=0;j--)
          if(arr[j+1]<arr[j]){
            [arr[j],arr[j+1]] = [arr[j+1],arr[j]]
          }else{
            break;
          }
    }
