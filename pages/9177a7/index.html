# 视频代码解析

<video
  id="video"
  src="https://dhxy.v.netease.com/2021/0311/92270351dc18546a2f36d9e4a33fa02d.mp4"
  controls = "controls"
  poster="images.jpg"  // 视频封面,未填写为第一帧截图。
  preload="auto"  //preload 属性规定是否在页面加载后载入视频。如果设置了 autoplay 属性，则忽略该属性。
  autoplay="autoplay"
  // 微博、微信，QQ内嵌、安卓浏览器可以自动播。
  // 未设置 muted情况下，IOS浏览器无法页面打开后自动播。
  // 设置 muted情况下，IOS浏览器页面打开后可以自动播。
  muted="muted"
  webkit-playsinline="true"
  playsinline="true"  // IOS属性。页面内嵌入播放。微博无论有无设置都是全屏播放。
  x-webkit-airplay="allow"
  x5-video-player-type="h5"  // 安卓属性。页面内嵌入播放。QQ内嵌不设置的话，层级最高。微博无论有无设置都是页面内嵌入播放。
  x5-video-player-fullscreen="true" // 安卓属性。全屏设置，设置为 true 是防止横屏
  x5-video-orientation="portraint" // 安卓属性。播放器的方向， landscape横屏，portraint竖屏，默认值为竖屏
  >
</video>

# JS 代码

解决 oppo reno4 点击播放视频-关闭视频-手机锁屏-手机开屏 按以上步骤执行后会出现没播放视频但出现视频音乐；

let browser;
browser = {
    versions: function () {
        var u = navigator.userAgent,
            app = navigator.appVersion;
        return { //移动终端浏览器版本信息
            isAppInside: /micromessenger/i.test(u.toLowerCase()) || /yixin/i.test(u.toLowerCase())||(u.toLowerCase().indexOf(' qq')>-1 && u.toLowerCase().indexOf('mqqbrowser')<0)
        };
    }()
};

//关闭视频时，做个判断，如果是浏览器打开的话，改变url参数
function rand(num) {
    return Math.floor(Math.random() * num) + 1;
}

if (!browser.versions.isAppInside) {
	window.history.pushState(null, null, window.location.href.split('?')[0] + "?_t=" + rand(9999999));
}

# 关于视频自动播放

由于 微博、微信（iso 版可 ready 触发 play）无法页面打开后自动播，IOS 浏览器非静音状态无法页面打开后自动播；QQ 内嵌、安卓浏览器可以自动播。不建议简单直接用 video 标签来做自动播放。

可使用 视频 H5 组件 来实现视频自动播放。http://webpack.nie.netease.com/index.html#pack=m_inline/basic/nieplayer.html

# 参考：直接 mp4（非组件）微信 IOS 版触发自动播放，安卓不支持

<script type="text/javascript" src="https://res.wx.qq.com/open/js/jweixin-1.2.0.js"></script>
<script>
    document.addEventListener("WeixinJSBridgeReady", function () {
        document.getElementById("videoID").play();
    }, false);
</script>
