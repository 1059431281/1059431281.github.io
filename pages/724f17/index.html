# useEventListener

本期介绍的是 vueuse 中的 useEventListener，属于 browser(浏览器)类别的 api。

# 什么是 useEventListener？

useEventListener 光看名字就知道这是一个有关监听，它就是 vueuse 为我们方便实使用事件监听器而而设计的 api，该 api 主要拥有的功能就是设置监听和 unmount 时自动卸载监听

# 官方的简单例子

# 1.监听 document 的可见行切换

import { useEventListener } from '@vueuse/core' useEventListener(document,
'visibilitychange', (evt) => { console.log(evt) })
1
2

效果就是页面可见行切换时（也可以说是当浏览器的标签页切换到后台或者切换到前台时到触发，通俗易懂点就是页面切换时触发）会执行参数输出 evt

# 2.给指定元素添加监听

import { useEventListener } from "@vueuse/core";

const element = ref<HTMLDivElement>();
useEventListener(element, "keydown", (e) => {
  console.log(e.key);
});

# 实现思路

# useEventListener 的基本思路

监听元素 ➡️ 监听的事件 ➡️ 事件的处理器+有关的 options 可选项

这是未封装前的基础流程操作，而在封装时我们就需要考虑更多，比如像是给监听元素默认值，让它为空时默认为监听 window。

# 如何自动卸载监听？

一般我们卸载监听都是在页面/组件卸取消挂载时候，那么我们在封装中也需要这样做，但因为我们很难在外部 js 中监听每个不同页面/组件的生命周期，所以我们可以采取用 watch 去监听元素来实现。

当元素生成或移动的时候都会触发 watch，届时我们再在内判断即可。

# 如何手动卸载监听？

将 api 内的卸载封装成函数当成返回值

# 参数详解

# target（可选）

**参数类型：**为 vueuse 自定义的 MaybeElementRef 元素类型，这里理解成传递一个元素就好

​ **目标：**所要绑定（监听）的目标元素

# event

监听的事件类型/名称

# listener

​ **参数类型：**参数为事件对象 event，返回值为 any 的函数

​ **目标：**监听触发时的事件处理器

# options（可选）

​ **参数类型：**可以为布尔值或监听事件的 options 对象

# 源码内一些小点解释

# 范型 E：

主要是限制 event 传递的参数值，通过 extends keyof WindowEventMap 来使 E 成为 WIndowEventMap 的子类

# 函数的 this 参数：

置于参数列表第一位，用来修改函数内的 this 指向（注：它是伪参数，编译成 js 不会有它）

# listener 的 ev 参数：

它对应的类型是通过 WindowEventMap[E]获取 event 事件对象

import type { Fn, MaybeRef } from "@vueuse/shared";
import { isString, noop, tryOnScopeDispose } from "@vueuse/shared";
import { watch } from "vue-demi";
import type { MaybeElementRef } from "../unrefElement";
import { unrefElement } from "../unrefElement";
import { defaultWindow } from "../_configurable";

interface InferEventTarget<Events> {
  addEventListener(event: Events, fn?: any, options?: any): any;
  removeEventListener(event: Events, fn?: any, options?: any): any;
}

export type WindowEventName = keyof WindowEventMap; // window中所有事件的接口对象(可用显示看看type test = Extract<keyof WindowEventMap, unknown>;)
export type DocumentEventName = keyof DocumentEventMap; // docment中所有事件的接口对象

//	定义传递的listener的类型，参数为event，返回值为空(Event是模仿js的ts内置对象)
export interface GeneralEventListener<E = Event> {
  (evt: E): void;
}

/**
 * 这里开始主要做的都是调用该api的不同参数的重载情况，实际上就是对各种不同模式的参数类型进行判断
 */
/**
 * Register using addEventListener on mounted, and removeEventListener automatically on unmounted.
 *
 * Overload 1: Omitted Window target
 *
 * @see https://vueuse.org/useEventListener
 * @param event
 * @param listener
 * @param options
 * 第一种重载是省略tartget参数，保留event/listener/optiuons的写法，返回值为函数类型
 * 关于
 */
export function useEventListener<E extends keyof WindowEventMap>(
  event: E,
  listener: (this: Window, ev: WindowEventMap[E]) => any,
  options?: boolean | AddEventListenerOptions
): Fn;

/**
 * Register using addEventListener on mounted, and removeEventListener automatically on unmounted.
 *
 * Overload 2: Explicitly Window target
 *
 * @see https://vueuse.org/useEventListener
 * @param target
 * @param event
 * @param listener
 * @param options
 * 第二种重载就不是省略tartget参数来，要传递，但传递的是要监听的window对象/窗口
 */
export function useEventListener<E extends keyof WindowEventMap>(
  target: Window,
  event: E,
  listener: (this: Window, ev: WindowEventMap[E]) => any,
  options?: boolean | AddEventListenerOptions
): Fn;

/**
 * Register using addEventListener on mounted, and removeEventListener automatically on unmounted.
 *
 * Overload 3: Explicitly Document target
 *
 * @see https://vueuse.org/useEventListener
 * @param target
 * @param event
 * @param listener
 * @param options
 * 第三种重载一样，只是改成docuemnt对象了而已
 */
export function useEventListener<E extends keyof DocumentEventMap>(
  target: Document,
  event: E,
  listener: (this: Document, ev: DocumentEventMap[E]) => any,
  options?: boolean | AddEventListenerOptions
): Fn;

/**
 * Register using addEventListener on mounted, and removeEventListener automatically on unmounted.
 *
 * Overload 4: Custom event target with event type infer
 *
 * @see https://vueuse.org/useEventListener
 * @param target
 * @param event
 * @param listener
 * @param options
 * 第四种重载就是比较常用的了
 */
export function useEventListener<Names extends string, EventType = Event>(
  target: InferEventTarget<Names>,
  event: Names,
  listener: GeneralEventListener<EventType>,
  options?: boolean | AddEventListenerOptions
): Fn;

/**
 * Register using addEventListener on mounted, and removeEventListener automatically on unmounted.
 *
 * Overload 5: Custom event target fallback
 *
 * @see https://vueuse.org/useEventListener
 * @param target
 * @param event
 * @param listener
 * @param options
 * 第五重载就是比较常用的了
 * 第一个参数自定义任意元素，且可用为空和undfined
 * 第二个参数是
 */
export function useEventListener<EventType = Event>(
  target: MaybeRef<EventTarget | null | undefined>,
  event: string,
  listener: GeneralEventListener<EventType>,
  options?: boolean | AddEventListenerOptions
): Fn;

export function useEventListener(...args: any[]) {
  let target: MaybeRef<EventTarget> | undefined;
  let event: string;
  let listener: any;
  let options: any;

  if (isString(args[0])) {
    [event, listener, options] = args;
    target = defaultWindow;
  } else {
    [target, event, listener, options] = args;
  }

  if (!target) return noop;

  let cleanup = noop;

  const stopWatch = watch(
    () => unrefElement(target as unknown as MaybeElementRef),
    (el) => {
      cleanup();
      if (!el) return;

      el.addEventListener(event, listener, options);

      cleanup = () => {
        el.removeEventListener(event, listener, options);
        cleanup = noop;
      };
    },
    { immediate: true, flush: "post" }
  );

  const stop = () => {
    stopWatch();
    cleanup();
  };

  tryOnScopeDispose(stop);

  return stop;
}
