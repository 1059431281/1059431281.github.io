// store/index.jsx
import { createStore } from "redux"

    default:

  }
}

export default createStore(todos)

# connect

接着，在Provider提供之后，我们要使用绑定在provider上的store，使用用connect去跟需要使用的文件进行连接。

它有两个参数，一个是关于state的函数，一个是关于dispatch的函数，具体使用看代码。

connect第一次调用会返回函数，这时候再调用一次把组件作为参数传入，关于store的东西会被绑定在该组件的props上。

import { connect } from "react-redux";

const mapStateToProps = state => {
  return {
    state: state
  }
}

const ADDCOUNTACTION = {
  type: "add"
}

const mapDispatchToProps = dispatch => {
  return {
    addCount: () => {
      // 因为这里的dispatch通过connect之后获取到的是store的dispatch，所以去调用等同于调用了store的dispatch
      dispatch(addCountAction)
    }
  }
}


function App(props){
  console.log(props) // {state:...,addCount:...}
  return (
  	<div></div>
  )
}

export default connect(mapStateToProps,mapDispatchToProps)(App)

这玩意可直接就是reddx的大升级了，相比redux无论是用法还是什么都更方便，而且集成一体，不需要再额外引入apply....这些玩意了。

# createSlice

接收一组 reducer 函数的对象，一个 slice 切片名和初始状态 initial state，并自动生成具有相应 action creator 和 action type 的 slice reducer。

// counterSlice
import { createSlice } from "@redux/toolkit";

export const counterSlice = createSlice({
  // name对应的就是我们构造的action所会对应的前缀，也就是counter/increment。
  name: "counter",
  // 初始化值。
  initialState: {
    value: 0
  },
  reducer: {
    increment: (state,action={payload:1}) =>{
      state.value += action.payload;
    }
  }
})

// 已经帮我们自动生成好action，将其直接导出到时配合useDispatch使用。
export const { increment } = counterSlice.actions;

// 导出我们的createSlice帮我们生成的reducer，供configureStore使用。
export default counterSlice.reducer;

# createReducer

createSlice会帮我们直接生成reducer，那我们要如何自己生成呢？就是靠它！

**官方的定义是：**帮你将 action type 映射到 reducer 函数，而不是编写 switch...case 语句。另外，它会自动使用 immer 库 (opens new window)来让你使用普通的 mutable 代码编写更简单的 immutable 更新，

// counterReducer.js
import { createReducer } from "@reduxjs/toolkit";

const initialState = { value:0 }

const counterReducer = createReducer(initialState,{
  increment: (state,action={payload:1})=>{
    state.value += action.payload;
    // 又或者
    // return {
    //  value: state.value += actionpayload
    // }
  }
})

export default counterReducer;

# createAction

**官方定义：**成给定 action type 字符串的 action creator 函数。该函数本身已定义了 toString()，因此可以代替常量类型使用。

// counterReducer.js
import { createReducer,createAction } from "@reduxjs/toolkit";

const initialState = {value:0}

export const increment = createAction('counter/increment');

const counterReducer = createReducer(initialState, {
  [increment]:(state)=>{
    // ...省略了
  }
  // 或者
  //[increment.type]:....
  // 跟上面的不同，

# configureStore

封装了createStore，简化了配置项，并且提供了一些县城的默认配置。它可以自动组合slice的reducer，可以添加容易redux。此外，默认情况下保护着redux-thunk，也就不需要我们自己引这东西，此外还开启了reduxdevetools扩展，调试更好。

// store/index.js

import { configureStore } from "@reduxjs/toolkit"
import counterreducer from "counterReducer.js"

export default configureStore({
  reducer: {
    counter: counterreducer
  }
})

# middleware(中间件)

reduxjs/toolkit已经做好了对在中间件的封装，不需要我们额外再去引入，只需要传递必要的参数就好。

这里先放一段官方的原文:

An optional array of Redux middleware functions

If this option is provided, it should contain all the middleware functions you want added to the store. configureStore will automatically pass those to applyMiddleware.

If not provided, configureStore will call getDefaultMiddleware and use the array of middleware functions it returns.

Where you wish to add onto or customize the default middleware, you may pass a callback function that will receive getDefaultMiddleware as its argument, and should return a middleware array.

大致的意思就是：

中间件的函数是存放在数组里面。

如果middleware这个参数有被填写，那它应该包含所有你想添加到store到中间件函数，configureStore会自动传入到applyMiddleware里面。

如果没有填写这个参数，它会调用getDefaultMiddleware，用中间件函数数组作为返回值。

如果你希望王默认的中间件中加入自定义的，你能通过一个回调函数，该回调函数会收到getDefaultMiddleware作为参数，然后我们应该返回一个中间件数组。

关于它最后还有一段话，就是你想了解更多middleware的参数，那就访问https://redux-toolkit.js.org/api/getDefaultMiddleware

// store/index.js

import { configureStore } from "@reduxjs/toolkit"
import counterreducer from "counterReducer.js"
import logger from "redux-logger"
import thunk from "redux-thunk"

export default configureStore({
  reducer: {
    counter: counterreducer
  },
  middleware:[logger,thunk]
})

# getDefaultMiddleware

这个是reduxjs/toolkit的默认中间件，上面的翻译中已经知道它会默认传递，不传参的话。那么，要如何拓展它呢？

// store/index.js

import { configureStore } from "@reduxjs/toolkit"
import counterreducer from "counterReducer.js"
import logger from "redux-logger"
import thunk from "redux-thunk"

export default configureStore({
  reducer: {
    counter: counterreducer
  },
  middleware:(getDefaultMiddleware)=>getDefaultMiddleware().concat(logger)
  // getDefaultMiddleware调用后会返回数组，这时候我们再用concat进行数组合并来进行拓展。
})
