/**
	下面这一段就不在详解，需要注意的它使用vueuse中的另外两个api，一个是用来绑定事件，一个是获取window的定义
*/
import { ref } from "vue-demi";
import type { ConfigurableEventFilter } from "@vueuse/shared";
import { useEventListener, defaultWindow } from "@vueuse/core";
import type { ConfigurableWindow, Position } from "@vueuse/core";

/**
 * 老样子用ts的写法定义该api能传参的类型，其次这里用了export是为了方便再其他文件也能引入类类型（import type {MouseOptions} from 'xxx'）
 *【重点】：这里他用extends继承了ConfigurableWindow和ConfigurableEventFilter类型，这里不懂的可以先跳到下面，有做类型接口展示
 * 在基础了前两者的基础上，这里还另外定义了四个属性，汉译分别是取值类型、是否启动触摸监听、重置、初值
 */
export interface MouseOptions
  extends ConfigurableWindow,
    ConfigurableEventFilter {
  /**
   * Mouse position based by page or client
   * 鼠标的位置是基于page还是client（page的位置是基于当前可视位置加上滚动条已滚动数据，clinet则不包含）
   *
   * @default 'page'
   * 默认值为page
   */
  type?: "page" | "client";
  /**
   * Listen to `touchmove` events
   * pc和移动端的鼠标事件不一样，这里是是否启用移动端触摸时监听
   *
   * @default true
   */
  touch?: boolean;

  /**
   * Reset to initial value when `touchend` event fired
   * 这是当移动端事件中的触摸松开时是否要初始化鼠标位置值
   *
   * @default false
   */
  resetOnTouchEnds?: boolean;

  /**
   * Initial values
   * 给予鼠标初始位置，应搭配resetOnTouchEnds使用
   */
  initialValue?: Position;
}

# ConfigurableWindow（可配置 window 项）

允许我们自定义 window 实例，例如用 iframe 嵌入时或者测试环境

简单解读一下就是我们可以传参更改 window 实例，不传则使用默认的 window。

这里的 window?:widnow 则是限制了参数应当为 window，否则为空**（这一段才是这个接口代码的作用，上面两行是应用场景解读）**

export interface ConfigurableWindow {
  /*
   * Specify a custom `window` instance, e.g. working with iframes or in testing environments.
   */
  window?: Window;
}

# 3

// 定义鼠标获值可用类型
export type MouseSourceType = "mouse" | "touch" | null;

/**
 * Reactive mouse position.
 *
 * @see https://vueuse.org/useMouse
 * @param options
 */
export function useMouse(options: MouseOptions = {}) {
  /**
  	可以看到默认值都是在这里设的，所以说其实是可以改的！但是....不会真的有人跑去改源码把...
  */

  const {
    type = "page",
    touch = true,
    resetOnTouchEnds = false,
    initialValue = { x: 0, y: 0 },
    window = defaultWindow,
    eventFilter,
  } = options;

  const x = ref(initialValue.x);
  const y = ref(initialValue.y);
  const sourceType = ref<MouseSourceType>(null);

  //pc端处理器
  const mouseHandler = (event: MouseEvent) => {
    if (type === "page") {
      x.value = event.pageX;
      y.value = event.pageY;
    } else if (type === "client") {
      x.value = event.clientX;
      y.value = event.clientY;
    }
    sourceType.value = "mouse";
  };
  // 重置鼠标位置为初始值
  const reset = () => {
    x.value = initialValue.x;
    y.value = initialValue.y;
  };
  //移动端处理器
  const touchHandler = (event: TouchEvent) => {
    if (event.touches.length > 0) {
      const touch = event.touches[0];
      if (type === "page") {
        x.value = touch.pageX;
        y.value = touch.pageY;
      } else if (type === "client") {
        x.value = touch.clientX;
        y.value = touch.clientY;
      }
      sourceType.value = "touch";
    }
  };

  // pc和移动端事件容器，判断是否有自定义事件过滤器，没有的话就直接用我们上面定义的处理器。有的话就使用定义的事件过滤器，将我们上面定义的处理作为参数传递给该事件流（过滤器）
  const mouseHandlerWrapper = (event: MouseEvent) => {
    return eventFilter === undefined
      ? mouseHandler(event)
      : eventFilter(() => mouseHandler(event), {} as any);
  };

  const touchHandlerWrapper = (event: TouchEvent) => {
    return eventFilter === undefined
      ? touchHandler(event)
      : eventFilter(() => touchHandler(event), {} as any);
  };

  // 多嘴一句，这里的passive是让默认事件顺从的一个设定，让监听的处理器和浏览器的默认行为能同事进行，从而避免卡顿
  if (window) {
    useEventListener(window, "mousemove", mouseHandlerWrapper, {
      passive: true,
    });
    useEventListener(window, "dragover", mouseHandlerWrapper, {
      passive: true,
    });
    if (touch) {
      useEventListener(window, "touchstart", touchHandlerWrapper, {
        passive: true,
      });
      useEventListener(window, "touchmove", touchHandlerWrapper, {
        passive: true,
      });
      // 这里就是绑定时候手指移开屏幕时要将position恢复到初始值了
      if (resetOnTouchEnds)
        useEventListener(window, "touchend", reset, { passive: true });
    }
  }

  return {
    x,
    y,
    sourceType,
  };
}

// 将返回值类型导出
export type UseMouseReturn = ReturnType<typeof useMouse>;
