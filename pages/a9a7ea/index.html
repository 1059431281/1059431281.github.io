# slots

# 建议方法：

componentFunc.call(this,{
  slots: {

  }

  slots: {
    default: [componentName,组件参数({{props:{}}})]

})

/**
 * @description: 混入插槽
 * @return {*}
 * @param {*} instance
 */
function mixinSlot(instance, slots) {
  slots = slots || {}
  for (const key in slots) {
    if (Array.isArray(slots[key])) {
      slots[key][1] = slots[key][1] || {}
      slots[key][1].on['Close'] = registerProp({ instance, visible: false })
      instance.$slots[key] = instance.$createElement(slots[key][0], slots[key][1])
    } else {
      instance.$slots[key] = instance.$createElement(slots[key])
    }
  }
}

# element二次封装

import { registerComponent } from './generate'
import { Dialog, Drawer } from 'element-ui'
// 为实现不同visible，采用逐渐增加到id数
const idNum = 1

const { ElDialog, ElDrawer } = registerComponent([Dialog, Drawer], 'multiple') // 多例

/**
 * @description: 标准化通用组件二级封装函数
 * @author: zxd
 * @return {*}
 * @param {*} func 不同组件组件类型有不同的函数
 * @param {*} options 组件参数
 * @param {*} slots 插槽有关，可传对象{插槽名：组件|[组件，插槽参数]}
 * @param {*} showMode 显示模式
 */
function normalizeComponent(func, options = {}, slots, showMode = 'overlay') {
  this.visible = true
  options.visible = true
  const events = {
    'update:visible': function(v) {
      this.visible = v
    }
  }
  options = {
    props: options,
    slots,
    showMode,
    events
  }
  const instace = func.call(this, options)
  return instace
}

export function openDialog(options, slots, showMode) {
  return normalizeComponent.call(this, ElDialog, options, slots, showMode)
}

export function openDrawer(options, slots, showMode) {
  return normalizeComponent.call(this, ElDrawer, options, slots, showMode)
}
