# 前言

基础算法的时间复杂度也就是速度不理想，不可能拿它来处理大量数据，为此使用的就是高级算法

# 快速排序

这一算法是处理大数据最为常见的算法之一，具体的思路就是分区治理，以递归的方式依次依次分解为较小和较大的子队列，直到子队列不可再划分即是排序好的数据。

function quickSort(arr){
  if(arr.length<=1){
    return arr;
  }
  let left = [];
  let right = [];
  let cur = arr.splice(0,1);
  for(let v of arr){
    if(v < cur) {
      left.push(v)
    }else{
      right.push(v)
    }
  }
  return quickSort(left).concat(cur,quickSort(right));
}

# 希尔排序

希尔排序是插入算法的改良

function shellSort(arr,gap) {
    console.log(arr)
    for(let i = 0; i<gap.length; i++) { 
        let n = gap[i]; 
        for(let j = i + n; j < arr.length; j++) {
            for(let k = j; k > 0; k-=n) { 
                if(arr[k] < arr[k-n]) {
                    [arr[k],arr[k-n]] = [arr[k-n],arr[k]];
                } else {
                    continue;
                }
            }
        }
    }
    return arr;
}
