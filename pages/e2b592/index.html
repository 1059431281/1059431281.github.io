# pdf和canvas相互转换

# canvas转换pdf

# 需要工具

  • html2canvas 将html页面转换为图片
  • jspdf 将base64转换为pdf https://github.com/parallax/jsPDF
let base64Canvas = ""
const { jsPDF } = jspdf
const pdf = new jsPDF('','pt','a4')
html2canvas(document.getElementById('test'),{       
  scale: 2, // 缩放值,大一点可以提高图片清晰度
  // canvas: canvas, //自定义 canvas
  // logging: true, //日志开关，便于查看html2canvas的内部执行流程
  // width, //dom 原始宽度
  // height: height,
  // useCORS: true, // 【重要】开启跨域配置
  // allowTaint: true,  
  // taintTest: false
}).then(canvas => {

  var contentWidth = canvas.width;
  var contentHeight = canvas.height;

  // a4纸的标准尺寸【595.28.841.89】
  // 计算出canvas与a4纸的宽比，接着再去乘以标准的a4高度，得出一页pdf要显示的canvas高度
  var pageHeight = ( contentWidth / 595.28 ) * 841.89;
  var leftHeight = contentHeight;

  var position = 0; // 页面偏移量

  var imgWidth = 595.28;
  var imgHeight = ( 595.28 / contentWidth ) * contentHeight;

  /**
     * pageHeight和imgheight的区别：
     * pageHeight得到的是，当我们的pdf转换为当前canvas时，对应的一页高度
     * imgheight得到的是，当我们按照a4纸的标准宽时，我们img的高度也要有相应变换
     * 两者比较明显的区别有，pageHeight是计算 contentWidth 与 592.28 之比，后者则是反过来。
     * 简单说，pageHeight就是将标准的a4转换为当前canvas的单位，而imgHeight就是将当前canvas转换为pdf的单位。
     * pageHeight的主要作用，是用来判断我们当前canvas转换之后的高度，是否超过了转化pdf后一页高度
     * 能看到，后续用leftHeight，也就是我们的canvas高度去跟pageHeight进行对比，如果小于，后面就达到可以在一页显示，如果大于要进行分页处理
    */

  base64Canvas = canvas.toDataURL('image/jpeg',2.0)

  if ( leftHeight < pageHeight ) {
    pdf.addImage(base64Canvas, "JPEG", 0, 0, imgWidth, imgHeight)
  } else {
    while( leftHeight > 0 ){
      pdf.addImage(base64Canvas, "JPEG", 0, position, imgWidth, imgHeight)
      leftHeight -= pageHeight;
      position -= 841.89;
      // 避免添加空白页
      if ( leftHeight > 0) {
        pdf.addPage();
      }
    }
  }
  pdf.save("content.pdf");
})

# pdf转换为canvas

# 需要工具

  • pdf.js (opens new window) 包含pdf.js和pdf.worker.js
document.getElementById('upload').onchange = function(e){

function createcanvas(url){
  const loadingTask = pdfjsLib.getDocument(url); // 读取文件，通过文件内的promise属性去获取读取文件后的数据
  loadingTask.promise.then(async (pdf)=>{
    console.log('pdf:', pdf);
    let tasks = Promise.resolve(null);
    /** pdf粥有numPages属性，代表页数
      * 去遍历页数，然后对遍历出来的页面进行渲染
      * 因为要保证渲染的页面顺序，这里使用promise去链式加载渲染
      * 通过getPage方法去获取对应页数的对象，然后通过该对象去获取视口对象，后面获取对应的渲染参数进行渲染了
      * 有一点要注意，同样的canvas不能重复被渲染，并且也不建议提前就在页面写着，最好的做法就是在每个页面渲染前创建canvas元素，去作为渲染的容器
      */
    for(let page = 1; page <= pdf.numPages ; page++) {
      tasks.then(
        pdf.getPage(page).then( pageData => {
          let scale = 1.5;
          let viewport = pageData.getViewport({ scale });

          let outputScale = window.devicePixelRatio || 1;

          let canvas = document.createElement("canvas")
          let context = canvas.getContext("2d");

          canvas.width = Math.floor(viewport.width * outputScale); // 注意，canvas的宽高是表示修改画布的大小都根据这个来定位
          canvas.height = Math.floor(viewport.height * outputScale);
          canvas.style.width = Math.floor(viewport.width) + 'px'; // 而style，就是纯粹画布和浏览器的渲染
          canvas.style.height = Math.floor(viewport.height) + 'px';

          let transform = outputScale !== 1
          ? [ outputScale, 0, 0, outputScale, 0, 0 ]
          : null;

          let renderContext = {
            canvasContext: context,
            transform: transform,
            viewport: viewport
          }

          pageData.render(renderContext)
          document.body.appendChild(canvas)
          if(page == pdf.numPages) test = null;
        })
      )
    }
  })
}

# 图片裁剪

参考代码：

因为是随手写的，所以起名时比较随意

<body>
  <div class="wrap" id="wrap">
    <canvas id="canvas-test" width="400" height="400"></canvas>
    <div class="cut"></div>
  </div>
  <input id="file" type="file">
  <button id="btn_cut">确定裁剪</button>
  <div class="wrap2">
    <canvas id="canvas-test2" width="400" height="400"></canvas>
  </div>
  <button id="btn_download">导出图片</button>

  <script>
    let cvs = document.getElementById("canvas-test")
    let cvs2 = document.getElementById("canvas-test2")
    let ctx = cvs.getContext("2d")
    let ctx2 = cvs2.getContext("2d")
    let wrap = document.getElementById("wrap")
    let cut = document.getElementsByClassName("cut")[0]
    let $btn_cut = document.getElementById("btn_cut")
    let $btn_download = document.getElementById("btn_download")

    var events = {touchstart:null,touchend:null,touchmove:null} // 用对象将函数存起来，为了让removeEventLisnter能删除到指定监听
    document.getElementById("file").onchange = function(e){
      cut.removeEventListener("touchstart",events.touchstart) // 在图片改变后，删除掉上一次监听，如果有的话（这里没做如果有的判断，单纯懒）。
      cut.removeEventListener("touchend",events.touchend)
      cut.removeEventListener("touchmove",events.touchmove)
      let file = e.target.files[0]
      let base64 = URL.createObjectURL(file) // 将file文件对象格式转换base64的形式，否则无法读取
      let img = new Image();
      img.src = base64;
      img.onload = function(){

        let 
          img_width = img.width, // 原图高度
          img_height = img.height, // 原图宽度
          cvs_width = cvs.offsetWidth, // canvas的现宽度
          cvs_height = cvs.offsetHeight;  // canvas的现高度

        let translateY = 0; // canvas的偏移量，默认是居中，只有当图片高度高于canvas高度时才需要使用。
        let oldY = 0; // 旧的y值，也就是鼠标滑动的前一个y值。
        let ratio = img_height / img_width // 原图高度与原图宽度之比。
        let width = img_width > cvs_width ? cvs_width : img_width ; // 当原图高度大于canvas的宽度，就用canvas的宽，反之用原图宽
        let height = width * ratio // 如果上面得到的宽度，根据高宽比，获取canvas要显示的高度
        let x = (cvs_width - width) / 2 ; // canvas的本身宽度减去canvas实际要显示的宽度，除以2，得到差值的一半；
        let y = (cvs_height -height) / 2; // 同理。不过额外补充一句，这里得到的y值就是我们canvas可以进行偏移的值
        let height_ratio = img_height / height // 获取原图高度跟计算出来的高度之比，用以我们后面裁剪用。
        let isDownload = false; // 判断是否有裁剪好的图片，否则不能导出图片
        let startime; // 用来存储鼠标按下的时间
        let timer = null;

        ctx.clearRect(0,0,img_width,img_height) // 清空画布。


        events.touchstart = function(e){ 
          startime =  e.timeStamp;
          oldY =   e.changedTouches[0].clientY;
          clearInterval(timer)
          timer = null;
        }
        events.touchend = function(e){
          let clientY = e.changedTouches[0].clientY;
          let direcUp = oldY > clientY;
          endTime = e.timeStamp;
          if(endTime - startime < 500 && Math.abs(oldY - clientY) > 2) {
            timer = setInterval(() => {
                if(direcUp){
                // 向上
                if(--translateY <= y) {
                  translateY = y;
                  clearInterval(timer)
                }else {
                  translateY -= 2;
                }
              }else{
                // 向下
                if(++translateY >= -y) {
                  translateY =  -y 
                  clearInterval(timer)
                }else {
                  translateY += 2; 
                }
              }
              cvs.style.transform = `translateY(${translateY}px)`
            }, 10 );        
          }       
        }    
        events.touchmove = function(e){
          let clientY = e.changedTouches[0].clientY;
          if(oldY > clientY ){
            // 向上
            if(--translateY <= y) {
              translateY = y
            }else {
              translateY -= 1;
            }
          }else{
            // 向下
            if(++translateY >= -y) {
              translateY =  -y 
            }else {
              translateY += 1;
            }
          }
          oldY = clientY;
          cvs.style.transform = `translateY(${translateY}px)`
        }
      

        if(height > 400) {
          cvs.style.marginTop = (y + 50) + "px" ; 
          /** 这里之所以要+50，是因为当大于400时，我们计算出来的是负数y，然后我们还想要他500高的wrap中居中，就要加上500-400=100的一半。
          *   比如442的高度，得到y值-21，然后再加上+50，就得到29，这样提供上边距29，就能在500高的容器中，让442高的canvas居中
          *   如果是大于500的，也同理。
          */
          cvs.height = height;
          cvs.style.height = height + "px"
          ctx.drawImage(img,0,0,img_width,img_height,x,0,400,height)          
          cut.addEventListener("touchstart",events.touchstart)
          cut.addEventListener("touchend",events.touchend)
          cut.addEventListener("touchmove",events.touchmove)
        }else {
          // 当canvas要显示的高度小于canvas本身高度时，直接给50px上边距，让我们canvas居中再wrap容器中
          cvs.style.marginTop = "50px" 
          // 这种我们通过y值得到的是，正数的y，通过这y确定我们绘画图片的位置
          ctx.drawImage(img,0,0,img_width,img_height,x,y,400,height)
        }


        $btn_cut.onclick = function(){
            let w = img_width
            let h = img_height + y*height_ratio*2; // 先通过y值，也就超出canvas展示的部分去乘以与原图的高度比例，得到转换为原图的高，然后被原图高减去，就等到要裁剪的高度(这里用加号，是因为这里高大于400时y时负数，相加才是减)
            ctx2.clearRect(0,0,400,400)
            ctx2.drawImage(img,0,(-translateY - y)*height_ratio,w,h,0,0,400,400)
            isDownload = true
        }
        $btn_download.onclick = function(){
          if(!isDownload) {
            alert("请先生成裁剪图片")
            return;
          }
          let base64 = cvs2.toDataURL('image/jpeg',2.0)
          let aLink = document.createElement("a")
          aLink.href = base64
          aLink.download = file.name
          aLink.click()
        }

      }
    }
  </script>
</body>
</html>

# 图片压缩

我们在上传图片给后端时，会出现图片过大的情况，这样会占用服务器空间，这种时候我们就可以要对上传的图进行压缩。

# 思路

通过canvas重绘，对质量和尺寸修改，最终转为blob传给后端

<body>
  <input type="file" id="file">
  <div id="wrap"></div>
  <script>
    document.getElementById("file").onchange = function(e){
      const file = e.target.files[0]
      // 相比URL.createObjectUrl,是异步的，并且在不用时会被清除，而不是等到页面卸载才清除
      const reader = new FileReader();
      reader.readAsDataURL(file)
      reader.onload = () => {
        const img = new Image();
        img.src = reader.result; // base64
        img.onload = () => {
          // 原图宽高
          const w = img.width;
          const h = img.height;

          const canvas = document.createElement('canvas');
          canvas.width = w * 0.5;
          canvas.height = h * 0.5;
          canvas.style.height = h +'px';
          canvas.style.width = w +'px';
          const ctx = canvas.getContext('2d');

          ctx.drawImage(img, 0, 0, w * 0.5, h * 0.5)

           canvas.toBlob((blob) => {
            const img2 = new Image();
            img2.src = window.URL.createObjectURL(blob)
            img2.onload = () => {
              document.getElementById('wrap').innerHTML = ''
              document.getElementById('wrap').appendChild(img)
              document.getElementById('wrap').appendChild(img2)
              console.log('压缩前图片大小:',file.size, '压缩后图片大小:',blob.size );
              console.log('压缩前图片宽:', w, '压缩后图片宽:', img2.width);
              console.log('压缩前图片高:', h, '压缩后图片高:', img2.height);
            }

    function dataURLtoBlob(dataurl) {
      const arr = dataurl.split(","),
        mime = arr[0].match(/:(.*?);/)[1],
        bstr = atob(arr[1]);
      let n = bstr.length;
      const u8arr = new Uint8Array(n);
      while (n--) {
        u8arr[n] = bstr.charCodeAt(n);
      }
      return new Blob([u8arr], { type: mime });
    }

    function getImgByteSize(data) {
      if (data) { // 获取base64图片byte大小
        const equalIndex = data.indexOf('=');  // 获取=号下标
        if (equalIndex > 0) {
          const str = data.substring(0, equalIndex); // 去除=号
          const strLength = str.length;
          const fileLength = strLength - (strLength / 8) * 2; // 真实的图片byte大小
          return fileLength >>> 0
        } else {
          const strLength = data.length;
          const fileLength = strLength - (strLength / 8) * 2;
          return fileLength >>> 0
        }
      } else {
        return null;
      }
    }
