组件函数指令生成器1.0

# 起因

随着项目愈发复杂，或者单文件涉及代码量过多，高复用性组件每次使用都需要去重新引入然后再去注册加以使用，数量少时还好，一旦多了操作起来会复杂，并且html区域会出现一些本不应该在中的组件标签代码。

除此之外，功能的唯一性也很关系，有时候想去调用起一个弹窗，不仅需要用变量去控制它，使用到的组件还需要放在html中不知什么地方，不能彻底将弹窗的全部放在调用弹窗的函数当中

# 优点

  1. 减少代码体积
  2. 使用方便
  3. 可读性好

# 使用方便

会用到函数式组件的地方，实际上就是一些需要通过【交互】来触发的组件，诸如弹窗、抽屉柜确认框等。

既然是要通过【交互】来触发的组件，那么在我们的文件中就会有对应的方法，例如open、close，或者其他通用的功能组件，也就是说文件中的 【方法】与【组件】一一对应。

不需要自己去引入注册，只需要将方法引入后直接用函数调用即可。

# 缺点

只适合悬浮在页面上的一类组件用，如果跟布局有关的还那样用，会适得其反。

# 使用方法

前言：推荐在文件夹下单独另起一个js管理组件

使用registerComponent方法

  • 第一个参数为文件对象，也可以是数组文件对象（如果是数组会返回包含$+文件名的对象）
  • 第二个参数为注册模式，可传'single'或'multipe'，分别是单例和多例，默认是single
// index.js
import { registerComponent } from '@/components/componentFunctional2/generate'

export const $dispatchNurse = registerComponent(require('./dispatchNurse.vue'))
export const $dispatchNurse_sure = registerComponent(require('./dispatchNurse_sure.vue'))

# 实现思路

  1. 既然要免去注册这一步骤，那么我们就需要自己去生成对应的【组件构造器】，然后再对组件进行挂载。
  2. 平时使用的脚手架，有关import的性质他们最终的做法是哪怕对应的文件经历过路由的离开和进去都会保持着最初访问import时的import状态，因此我们在实现时要做到当引用组件的父组件销毁后自动销毁组件
  3. 要实现成模块导出，并且等到第一次交互触发时才去生成构造器，否则就像全局注册一样。

# 外部使用的注册组件方法

/**
 * @description: 注册组件
 * @author: zxd
 * @return {*}
 * @param {*} requireComponent 组件文件对象或数组
 * @param {*} mode 采用的组件模式 单例/多例 single/multiple
 */
export const registerComponent = function(requireComponent, mode = 'single') {
  // // 如果是数组，就依次去遍历然后进行注册
  if (Array.isArray(requireComponent)) {
    const componentObj = {}
    requireComponent.forEach(($vnode) => {
      // 如果是用require，那么就需要default才能获取到具体组件对象
      $vnode = $vnode.default || $vnode
      const componentName = $vnode.name || $vnode.__file.replace(/.*\//, '').replace(/\.\w+$/, '')
      componentObj['$' + componentName] = initComponent.call(this, option, $vnode)
    })
    return componentObj
  } else {
  // 如果是用require，那么就需要default才能获取到具体组件对象
    const $vnode = requireComponent.default || requireComponent
    return componentMode[mode]($vnode, mode)
  }
}

# 初始化组件方法

function initComponent(option, $vnode, mode, instance) {
  const F_this = this // 调用该函数所用的this，需要使用该函数的时候用call或者bind之类去改变this
  // 这里主要是用在判断是不是首次调用该函数，因为在路由离开时会清除所有组件包括容器元素的下的元素，因此这里判断如果是当前路由首次就去执行一次挂载
  if (!containerShow) {
    containerShow = true
    document.body.appendChild(container)
  }
  if (!instance) {
    // 这里的this，就是该函数被应用时所在的vue文件下
    // 这个动作必须放在实例构造器之前，因为inject/provide的步骤是在created之前
    if (F_this && F_this._provided) {
      $vnode.beforeCreate.unshift(function() {
        // 这里是$vnode实例后的this，如果if条件内的this不同
        this._provided = F_this._provided
      })
    }
    const Constructor = Vue.extend({ ...$vnode, router })

    instance = new Constructor()

    instance._destory = () => {
      instance.close ? instance.close() : ''
      instance.$destroy()
      instance = null
      instaceCur = null
      if (containerShow) {
        container.remove()
        containerShow = false
      }
    }
    // 监听父组件销毁时，调用组件销毁方法
    F_this.$once('hook:beforeDestroy', instance._destory)
  }
  return initInstance.call(this, option, mode, instance)
}

# 初始化组件实例

const defaultShowMode = {
  single: 'cover',
  multiple: 'overlay'
}

const initInstance = ({ props, option, events, showMode } = {}, mode, instance) => {
  showMode = showMode || defaultShowMode[mode]
  initData(option)
  registerProp(props)
  // 如果新实例和最后使用的实例相同则不需要去进行挂载，直接返回实例子就好
  if (instaceCur && instaceCur._uid === instance._uid) return instance
  if (showMode === 'cover') {
    // 如果是要覆盖的显示方式，将显示组件的容器元素置空，并且有正在使用的实例便使用close关闭（如果有close方法的话，这里大致就是弹窗内必有的close方法）
    container.innerHTML = ''
    instaceCur && instaceCur.close ? instaceCur.close() : ''
    // 单例如果被挂着过一次，那么事件也应当注册过，考虑到实际业务需求，没必要再帮他重新去注册事件了(可以使用逃生舱，每次都强行去注册一遍)
    !instance.$el || ESCAPE_POD.registerEvent ? registerEmit(instance, events) : ''
  }
  if (showMode === 'overlay') registerEmit(instance, events)
  // 这里是用来判断实例是否有挂载过，如果挂载过就会有$el,没有则进行挂载(实际上，这里的判断更大哦是针对单例的情况)
  !instance.$el ? instance.$mount() : ''

  // 将现在也就是最后使用的组件实例存储起来
  instaceCur = instance
  container.appendChild(instance.$el)
  // 最后将实例返回，如果引用组件的文件用变量去存储它，就可以随意使用它了
  return instance
}

# 初始化组件实例的data函数

function initData(instance, option) {
  option = option || {}
  // 因为可能会有参数变化，所以最好每次打开都重新重置下参数，并且用深拷贝防止与外界产生联系
  for (const key in option) {
    if (option[key] && option[key] instanceof Object && instance[key]) {
      if (option[key] instanceof Array) {
        instance[key] = JSON.parse(JSON.stringify(option[key]))
      } else {
        Object.assign(instance[key], JSON.parse(JSON.stringify(option[key])))
      }
    } else {
      instance[key] = option[key]
    }
  }
}

# 注册props的函数

function registerProp(instance, props) {
  props = props || {}
  // 优化点，使用vue自带的缓存key数组
  for (const key in props) {
    if (props[key] !== instance._props[key]) {
      instance._props[key] = props[key]
    } else {
      instance.$set(instance._props, key, props[key])
    }
  }
}

# 注册事件的函数Emit

function registerEmit(instance, events) {
  events = events || {}
  for (const key in events) {
    instance.$on(key, events[key])
  }
}
