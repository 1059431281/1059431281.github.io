# useVModel

本期介绍的是 vueuse 中的 useVmodel，接下来就跟随文章一起学习它的源码吧！

# 什么是 useVModel？

useVModel 是一个用来简化我们做一些类似弹窗组件时的值传递的 api，下面就用一个简单的例子来方便理解下:（vue3 写法）

<!-- 父组件 -->
<template>
  <div>
    <formModel v-model:name="formData.name"></formModel>
    {{ formData }}
  </div>
</template>

<script lang="ts">
import { reactive } from "vue-demi";
import formModel from "xxx";
export default {
  components: {
    formData,
  },
  setup() {
    const formData = reactive({
      name: "名字",
    });
    return {
      formData,
    };
  },
};
</script>

<!-- 子组件 -->
<template>
  <div>
    <form>
      name:
      <input v-model="_name" />
    </form>
  </div>
</template>
<script lang="ts" setup>
import { useVModel } from "../utils/useVmodel";
import { defineEmits, defineProps } from "vue-demi";
const props = defineProps({
  name: String,
  sex: String,
});
const emit = defineEmits(["update:name"]);
const _name = useVModel(props, "name", emit);
</script>

我们这里把子组件理解成一个填写表单的弹窗，在父组件引入使用。

我们可以看到，我们在子组件使用 useVModel 返回的数据用来展示，因为 useVModel 返回的数据是响应式，现在先让我回过头看看不用 useVModel 的写法。

<!-- 子组件 -->
<template>
  <div>
    <form>
      name:
      <input v-model="props.name" />
    </form>
  </div>
</template>
<script lang="ts" setup>
import { useVModel } from "../utils/useVmodel";
import { defineEmits, defineProps } from "vue-demi";
const props = defineProps({
  name: String,
});
const emit = defineEmits(["update:name"]);
emit("update:name", 2);
</script>

这里的写法还是比较草率的，正常来讲我们会用 computed 或 watch 进行处理，在里面进行 emit 调用。总而言之，useVMode 就是省略下了 computed/watch 封装和 emit 的调用。

好了，既然现在理解了 useVMode 的使用场景，那么下面就来看看源码。

# 源码分析

就不一段段解析了，我在源码上加上自己的注释（英文是源码本身的注释）进行解释

/**
	记录点·unwrapRef需要跟随ts单独出个文章
	vue-demi跟单纯的vue不同，它里面包含vue2和vue3的安排，也就是一个同事兼容vue2与vue3的第三方包
*/
import type { UnwrapRef } from "vue-demi";
import { computed, getCurrentInstance, isVue2, ref, watch } from "vue-demi";

export interface VModelOptions {
    /**
     * When passive is set to `true`, it will use `watch` to sync with props and ref.
     * Instead of relying on the `v-model` or `.sync` to work.
     *
     * @default false
     *简单解释就是讲passive设置为true的时候弃用vue3的v-model或vue2的.sync的同步工作语法糖，改用watch的方法去同步父组件传递过来的props和返回给子组件使用ref（就是把props内要使用的变量重新定义成当前this下的ref(相应数据)）
     * 默认值为false
    */
    passive?: boolean;
    /**
     * When eventName is set, it's value will be used to overwrite the emit event name.
     *
     * @default undefined
     * 这个简单，就是如果你有传递事件名的话，这个事件名将会被设置后续会使用emit上
     * 默认值为undefined
     */
    eventName?: string;
    /**
     * Attempting to check for changes of properties in a deeply nested object or array.
     *
     * @default false
     * 是否对对象或数组进行深层监听（例如：{a:{achild:12}},一般改变achild是不会被监听到，值会监听一整个a是否改变）
     * 默认值为false
     */
    deep?: boolean;
}

/**
 * Shorthand for v-model binding, props + emit -> ref
 * v-model绑定方式的缩写
 *
 * @see https://vueuse.org/useVModel
 * @param props
 * @param key (default 'value' in Vue 2 and 'modelValue' in Vue 3)
 * vue2和vue3的key默认值是不一样的，vue2中默认值是v-model:value，vue3是v-model:modelValue
 * @param emit
 */

//这里涉及到ts的语法，它先使用泛型，也就是<>对一些函数内会用的类型进行预设
export function useVModel<
    P extends object,//要求类型P是object的子类
    K extends keyof P,//要求类型K是P的子类（这里深入一点讲是{name|sex}的子类，在P是object的子类的基础上，P也有自己包含的类型，例如上面例如中的props值，那么P的类型内就用{name|sex}，也就是说如果传入的K类型的值不是name或者sex就是错误的。这里用到ts的keyof实现了这一点）
    Name extends string//要求类型Name是string的子类
>(
    props: P,//要求props的参数类型为P
    key?: K,//可选参数，表示参数类型可以是K，当然也可以是其他甚至没有
    emit?: (name: Name, ...args: any[]) => void,//可选参数，表示参数类型可以是对应形式的函数类型(这里写的是Name的类型，其实也就是字符串类型，然后后面是任意可选参数，最后返回值为空)，当然也可以是其他甚至没有
    options: VModelOptions = {}//这里是选项参数，可选的，对应的参数要求根据上面的接口类型VModelOptions
) {
    const { passive = false, eventName, deep = false } = options;

  	//getCurrentInstance一个专门给特殊场景使用的api，可以把它理解成或者内部组件的this，但不推荐用来当作是组合式api的场景。附上官方链接(https://v3.cn.vuejs.org/api/composition-api.html#getcurrentinstance)
    const vm = getCurrentInstance();
    // @ts-expect-error mis-alignment with @vue/composition-api
  	/**
  	 * 这里从左边到右三个emit分别是有传emit时就直接获取，没有的话如果是vue3就用vm.emit获取，vue2的话就是vm.$emit.bind(vm)获取
  	 * 【注1：这里的场景是在内部组件有声明但没传递的情况】
  	 * 【注2：vue3的emit是直接.emit，而vue2需要加上$】

  	 * 关于event，是我们最终会用到的事件名，类型是字符串或undfined没什么好说。
  	 * eventName是从第四个可选参数中获得，如果有的话
    */
    const _emit = emit || vm?.emit || vm?.$emit?.bind(vm);
    let event: string | undefined = eventName;

    /**
     * 这里是没有传key的情况下处理，分为vue2和vue3两种处理方法
     * vue2：关于这里的vm.proxy.options.model获取可以去看看vue2.2起支持的自定义model，原理出自这（https://cn.vuejs.org/v2/api/#model）
     * 如果有定制value就拿定制的，如果没就用vue2中默认值value给key起名(上面说过)
     * 同理，如果有定制的event就拿定制，没有就用vue2中v-model语法糖缩写的input起名(vue2的默认形式就是v-model:value="默认值",@input="默认值()"。又记录了一边—— ——)

     * vue3: vue3的就比较简单了，v-model缩写的默认值是modelValue(v-model:modelValue="默认值"。再说一遍，vue2的是value)
     * 至于这里为什么不给event设置默认值，因为代码下面对event的最终获取是用update:xx的，跟vue3缩写前的@update:xx不谋而合（vue2是@input:xx）
    */
    if (!key) {
        if (isVue2) {
            const modelOptions = vm?.proxy?.$options?.model;
            key = modelOptions?.value || ("value" as K);
            if (!eventName) event = modelOptions?.event || "input";
        } else {
            key = "modelValue" as K;
        }
    }

    event = eventName || event || ``update:${key}``;

	/**
	 * 这里就是上面替到的，要用语法糖的同步功能还是自己用watch去实现了
	 * 说实话我也不知道这里为什么要区分两种，盲猜是想用检测深层嵌套功能
	*/
    if (passive) {
  	/**
  	 * 使用ref创建响应式数据props[key](按例子来就是props.name)。多说一句，key这个东西有可能为空，在ts中会判断它不能给到props，所以在后面加上感叹号，如果key为空就会抛出断言失败(断言就是上面用的as)
  	*/
        const proxy = ref<P[K]>(props[key!]);

  	/**
  	 * 猜猜这里为什么要用两次watch？
  	 * 第一个watch，是用来监听props[key]这个值的变化；第二个watch，是用来监听proxy(上面创的响应数据)。
  	 * 这里再说清楚一点，第一个watch监听的是父组件那边的变化后我们在子组件也做更新，第二个watch是监听子组件上数据的改变然后用emit把父组件的改了
  	*/
        watch(
            () => props[key!],
            (v) => (proxy.value = v as UnwrapRef<P[K]>)
        );

        watch(
            proxy,
            (v) => {
                if (v !== props[key!] || deep) _emit(event, v);
            },
            {
                deep,
            }
        );

        return proxy;
    } else {
  		/**
  		 * 正常情况下最终都会入到这个作用域
  		 * 用computed方法，这没什么好说，唯一需要注意的地方就是用泛型<p[K]>定义了返回类型
  		 * set方法就是触发了emit，照上面的例子来就是触发了emit("update:name",新值)
  		 * 多说一句，这里返回的是computed，而get会返回最终的值，而之所以最终uVModel返回的值是响应式的就是利用了computed 的特性
  		*/
        return computed<P[K]>({
            get() {
                return props[key!];
            },
            set(value) {
                _emit(event, value);
            },
        });
    }
}
