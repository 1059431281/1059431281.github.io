实际遇到的一些小问题

# 只允许数字和字母

/[^A-Za-z0-9]/

# 只需要整数数字

/^(?:0|(?:-?[1-9]\d*))$/

# 正整数，首位可以为0，但为0时不能跟有数字

^(0|[1-9]\d*)$

# 正整数，首位不为0

/^\+?[1-9]\d*$/

# 负整数，首位不为0

/^-[1-9]\d*$/

# 拓展

js的replace去掉0开头并且后面跟有字符的字符串的0，以及变为隶属于正整数的整数

   function formatPositiveInteger(str) {
      str += ''
      str = str.replace(/(\D|-)/g, '')// 去除非数字和负号
      if (/^0+./.test(str)) {
        // 使用replace方法替换开头的0，但保留单个的0
        str = str.replace(/^0+(?=.)/, '')
        console.log('str2', str)
      }
      return Number(str)
    },
